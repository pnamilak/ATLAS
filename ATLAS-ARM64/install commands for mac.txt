install commands for mac


rm -rf build dist ATLAS.spec

pyinstaller --noconfirm --clean --windowed \
  --name "ATLAS" \
  --hidden-import waitress \
  --hidden-import webview \
  --add-data "app/templates:app/templates" \
  --add-data "app/static:app/static" \
  --add-data "schema.sql:." \
  launcher_desktop.py


ls -la dist/ATLAS.app/Contents/Frameworks | grep schema



./dist/ATLAS.app/Contents/MacOS/ATLAS

lsof -nP -iTCP:5050 -sTCP:LISTEN

cat ~/Library/Application\ Support/ATLAS/logs/launcher.log



brew install create-dmg

mkdir dmg
cp -R dist/ATLAS.app dmg/

create-dmg \
  --volname "ATLAS" \
  --window-pos 200 120 \
  --window-size 800 400 \
  --icon-size 120 \
  --app-drop-link 600 185 \
  ATLAS.dmg \
  dmg/
